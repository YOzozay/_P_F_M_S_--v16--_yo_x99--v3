function doGet(e) {
  if (!e) return output({ error: "No event object" });
  const action = e.parameter?.action;

  if (action === "summary")               return output(getMonthlySummary(e.parameter.payMonth));
  if (action === "yearSummary")           return output(getYearSummary(e.parameter.year));
  if (action === "getExpenses")           return output(getExpenses(e.parameter.payMonth));
  if (action === "getWorklogs")           return output(getWorklogs(e.parameter.payMonth));
  if (action === "getExpenseMeta")        return output(getExpenseMeta());
  if (action === "getCreditSummary")      return output(getCreditSummary());
  if (action === "getCreditStatement")    return output(getCreditStatement(e.parameter.card_id, e.parameter.year, e.parameter.month));
  if (action === "getDebts")              return output(getDebts());
  if (action === "getInstallments")       return output(getInstallments());
  if (action === "getFixedList")          return output(getFixedList());
  if (action === "getConfig")             return output(getConfig());
  if (action === "getSalaryHistory")      return output(getSalaryHistory());
  if (action === "getUpcomingInstallments") return output(getUpcomingInstallments());
  if (action === "getCreditCards")        return output(getCreditCards());
  if (action === "getCarLoans")           return output(getLoans("car_loan"));
  if (action === "getHomeLoans")          return output(getLoans("home_loan"));

  return output({ status: "API running" });
}

  if (data.action === "addFixedExpense")           return addFixedExpense(data);
  if (data.action === "deleteFixedExpense")        return deleteFixedExpense(data);
  if (data.action === "toggleFixedExpense")        return toggleFixedExpense(data);
  if (data.action === "updateConfig")              return updateConfig(data);
  if (data.action === "addSalaryHistory")          return addSalaryHistory(data);
  if (data.action === "addDebt")                   return addDebt(data);
  if (data.action === "deleteDebt")                return deleteDebt(data);

function round2(n) {
  return Math.round((n + Number.EPSILON) * 100) / 100;
}

function fmt_date(d) {
  return Utilities.formatDate(new Date(d), "Asia/Bangkok", "yyyy-MM-dd");

function getYearSummary(yearStr) {
  const year = Number(yearStr);
  const ss = SpreadsheetApp.openById(SHEET_ID);
  
  const rawData = {
    cfg: getConfig(),
    salRows: ss.getSheetByName("salary_history").getDataRange().getValues().slice(1),

    instRows: ss.getSheetByName("credit_installments").getDataRange().getValues().slice(1)
  };

  const results = [];
  for (let m = 1; m <= 12; m++) {
    const s = getSummaryForMonthCore(year, m, rawData);
    results.push({ payMonth: s.payMonth, netIncome: s.netIncome || 0, totalExpenses: s.expenses?.totalExpenses || 0, netBalance: s.netBalance || 0 });
  }
  return results;
}

function getSummaryForMonthCore(year, month, rawData) {
  const { cfg, salRows, workLogs, expenses, fixedEx, loans, instRows } = rawData;
  const { startDate, endDate } = getPayPeriod(year, month);

  // --- 1. คำนวณรายได้ และ OT ---
  let workDays = 0, otMealDays = 0, holidayHours = 0, ot15 = 0, ot3 = 0;
  workLogs.forEach(r => {
    const d = new Date(r[1]);
    if (d >= startDate && d <= endDate) {
      workDays++;
      const h = Number(r[2]) || 0;
      const h15 = Number(r[3]) || 0;
      const h3  = Number(r[4]) || 0;
      holidayHours += h;
      ot15 += h15;
      ot3  += h3;
      if ((h15 + h3) >= cfg.ot_meal_threshold_hours) otMealDays++;
    }
  });

  let monthlySalary = 0;
  let sortedSal = [...salRows].sort((a,b) => new Date(a[0]) - new Date(b[0]));
  sortedSal.forEach(r => { if (new Date(r[0]) <= endDate) monthlySalary = Number(r[1]); });

  const hourlyRate    = (monthlySalary / cfg.salary_divisor_days) / cfg.work_hours_per_day;
  const otPay         = round2((holidayHours * hourlyRate * cfg.ot_multiplier_1) + (ot15 * hourlyRate * cfg.ot_multiplier_1_5) + (ot3 * hourlyRate * cfg.ot_multiplier_3));
  const mealNormal    = workDays * cfg.meal_normal_per_day;
  const mealOt        = otMealDays * cfg.meal_ot_per_day;
  const fuel          = workDays * cfg.fuel_per_day;
  const diligenceAllowance = cfg.diligence_allowance || 0;
  
  const grossIncome   = round2(monthlySalary + otPay + mealNormal + mealOt + fuel + diligenceAllowance);
  const totalGross    = grossIncome;
  
  const ssBase         = Math.min(monthlySalary, cfg.social_security_max_base || 15000);
  const socialSecurity = Math.min(750, round2(ssBase * (cfg.social_security_rate || 0.05)));
  const studentLoan   = cfg.student_loan_fixed || 0;
  const totalDeduction = socialSecurity + studentLoan;
  const netIncome      = round2(grossIncome - totalDeduction);

  // --- 2. คำนวณรายจ่าย ---
  let varTotal = 0;
  const byCategory = {};
  expenses.forEach(r => {
    const d = new Date(r[1]);
    if (d >= startDate && d <= endDate) {
      const amt = Number(r[3]) || 0;
      varTotal += amt;
      byCategory[r[2]] = (byCategory[r[2]] || 0) + amt;
    }
  });
  
  let fixTotal = 0;
  const fixItems = [];
  fixedEx.forEach(r => {
    const active = String(r[5]).toLowerCase() === "true";
    const start = r[3] ? new Date(r[3]) : null;
    const end   = r[4] ? new Date(r[4]) : null;
    if (!active || !start) return;
    if (start <= endDate && (!end || end >= startDate)) {
      const amt = Number(r[2]) || 0;
      fixTotal += amt;
      fixItems.push({ name: r[1], amount: amt });
    }
  });

  const totalExpenses = round2(varTotal + fixTotal);

  // --- 3. หนี้สิน (เก็บเฉพาะยอดสำหรับเดือนนั้นๆ) ---
  let totalLoanMonthly = 0;
  loans.forEach(r => {
    // r[2] is type, r[6] is monthly_due, r[14] is status
    if (r[14] === "active" && (r[2] === "car_loan" || r[2] === "home_loan")) {
      totalLoanMonthly += (Number(r[6]) || 0);
    }
  });

  let totalCreditMonthly = 0;
  instRows.forEach(r => {
     // r[5] is due_date, r[4] is amount
     const d = new Date(r[5]);
     if (d >= startDate && d <= endDate) {
       totalCreditMonthly += (Number(r[4]) || 0);
     }
  });

  const totalMonthlyDebt = round2(totalLoanMonthly + totalCreditMonthly + fixTotal);
  const dtiPercent = grossIncome > 0 ? round2((totalMonthlyDebt / grossIncome) * 100) : 0;

  // --- 4. ส่งผลลัพธ์ ---
  return {
    payMonth: `${year}-${String(month).padStart(2, "0")}`,
    period: { startDate: fmt_date(startDate), endDate: fmt_date(endDate) },
    income: { 
      monthlySalary, holidayHours, ot15, ot3, 
      otPay, mealNormal, mealOt, fuel, 
      diligenceAllowance, grossIncome, totalGross,
      social_security: socialSecurity
    },
    deductions: { socialSecurity, studentLoan, totalDeduction },
    netIncome,
    debtBreakdown: { 
      loans: round2(totalLoanMonthly), 
      credits: round2(totalCreditMonthly), 
      fixed: round2(fixTotal), 
      totalDebt: totalMonthlyDebt, 
      dti: dtiPercent 
    },
    expenses: { 
      variable: { total: round2(varTotal), byCategory }, 
      fixed: { total: round2(fixTotal), items: fixItems }, 
      totalExpenses 
    },
    netBalance: round2(netIncome - totalExpenses)
  };
}

function payDebt(data) {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const sheet = ss.getSheetByName("debts");
  const rows = sheet.getDataRange().getValues();
  const payAmt = Number(data.amount);

  for (let i = 1; i < rows.length; i++) {
    if (String(rows[i][0]).trim() !== String(data.debt_id).trim()) continue;
    if (rows[i][7] === "closed") return output({ error: "Debt already closed" });

    const remaining = Number(rows[i][5]) || 0;
    if (payAmt > remaining) return output({ error: "Payment exceeds remaining amount" });

    const newRem = round2(remaining - payAmt);
    sheet.getRange(i + 1, 6).setValue(newRem);
    if (newRem <= 0) sheet.getRange(i + 1, 8).setValue("closed");

    ss.getSheetByName("expenses").appendRow([
      Utilities.getUuid(), new Date(), "debt_payment", payAmt,
      "Transfer", `Debt: ${rows[i][1]}`, data.debt_id, "debt", new Date()
    ]);
    return output({ status: "success", remaining: newRem });
  }
  return output({ error: "Debt not found" });
}

// 2. ฟังก์ชันบันทึกสัญญา (แก้ปัญหาเรื่องยอดคงเหลืองวดบ้านไม่ตรง)
// ==========================================
function addLoan(data) {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const loanSheet = ss.getSheetByName("loans");
  const paySheet = ss.getSheetByName("loan_payments");

  const id = Utilities.getUuid();
  const totalAmt = Number(data.total_amount) || 0;
  const monthly = Number(data.monthly_due) || 0;
  const paidCount = Number(data.paid_installments) || 0; 

  // ✅ ถ้าผู้ใช้กรอกยอดคงเหลือปัจจุบันมา ให้ใช้ยอดนั้นเลย ถ้าไม่กรอกก็ให้หักลบแบบรถยนต์ (Flat Rate) ปกติ
  let currentRemainingAmt = totalAmt;
  if (data.remaining_amount && Number(data.remaining_amount) > 0) {
    currentRemainingAmt = Number(data.remaining_amount);
  } else {
    currentRemainingAmt = totalAmt - (paidCount * monthly);
  }

  // ✅ คำนวณเดือนที่เหลือ (เอายอดคงเหลือ หารด้วย ค่างวดต่อเดือน)
  const remainingMonths = monthly > 0 ? Math.ceil(currentRemainingAmt / monthly) : 0;
  const totalInst = paidCount + remainingMonths; // รวมจำนวนงวดทั้งหมด

  // บันทึกสัญญากลักลงชีต loans 
  loanSheet.appendRow([
    id, data.name, data.loan_type, data.lender || "", totalAmt, 0, monthly, 
    Number(data.due_day) || 1, totalInst, paidCount, currentRemainingAmt, remainingMonths, 
    data.start_date ? new Date(data.start_date) : new Date(), 
    data.end_date ? new Date(data.end_date) : "", "active", new Date()
  ]);

  // สร้างประวัติย้อนหลังลงชีต loan_payments (เพื่อให้หน้า Dashboard ไม่ว่างเปล่า)
  if (paidCount > 0 && paySheet) {
    const startDate = data.start_date ? new Date(data.start_date) : new Date();
    // ถ้าเป็นบ้าน เราไม่รู้ว่าอดีตหักต้นไปเท่าไหร่ (เพราะมีดอกเบี้ยผสม) เราตีรวมไปก่อนเพื่อไม่ให้ระบบ Error
    const pastPrincipalPerMonth = data.loan_type === "home_loan" 
      ? round2((totalAmt - currentRemainingAmt) / paidCount) 
      : monthly;

    for (let i = 0; i < paidCount; i++) {
      let payDate = new Date(startDate);
      payDate.setMonth(payDate.getMonth() + i); 
      
      paySheet.appendRow([
        Utilities.getUuid(), id, monthly, pastPrincipalPerMonth, 0, payDate, i + 1, new Date()
      ]);
    }
  }

  return output({ status: "success", id: id });
}

function payLoan(data) {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const sheet = ss.getSheetByName("loans");
  const paySheet = ss.getSheetByName("loan_payments");

  if (!sheet) return output({ error: "loans sheet not found" });

  const loanId  = String(data.loan_id || data.id || "").trim();
  const loanType = String(data.loan_type || "").toLowerCase().trim();
  
  // ✅ ตรวจสอบยอดจากหน้าบ้าน
  const payAmt  = Number(data.amount);
  const principal = data.principal !== undefined ? Number(data.principal) : payAmt;
  const interest  = data.interest !== undefined ? Number(data.interest) : 0;
  
  if (!payAmt || payAmt <= 0) return output({ error: "Invalid amount" });

  const rows = sheet.getDataRange().getValues();
  for (let i = 1; i < rows.length; i++) {
    if (String(rows[i][0]).trim() !== loanId) continue;
    if (String(rows[i][2]).toLowerCase().trim() !== loanType) continue;
    if (rows[i][14] === "closed") return output({ error: "Loan already closed" });

    const remAmt  = Number(rows[i][10]) || 0;
    const remMon  = Number(rows[i][11]) || 0;
    const paidInst = Number(rows[i][9]) || 0;

    // ✅ ถ้างวดบ้าน ให้หักลบยอดหนี้ด้วย "เงินต้น" ถ้างวดรถหักด้วย "ยอดจ่ายเต็ม"
    const deductAmount = loanType === "home_loan" ? principal : payAmt;

    if (deductAmount > remAmt + 1) return output({ error: "เงินต้นที่ชำระเกินยอดหนี้คงเหลือ" });

    const newRemAmt = round2(Math.max(0, remAmt - deductAmount));
    const newRemMon = Math.max(0, remMon - 1);
    const newPaidInst = paidInst + 1; 

    // 1. อัปเดตข้อมูลสัญญาในชีต loans
    sheet.getRange(i + 1, 11).setValue(newRemAmt);   // K: remaining_amount
    sheet.getRange(i + 1, 12).setValue(newRemMon);   // L: remaining_months
    sheet.getRange(i + 1, 10).setValue(newPaidInst); // J: paid_installments
    if (newRemAmt <= 0) sheet.getRange(i + 1, 15).setValue("closed");  // O: status

    // 2. บันทึกลง expenses เพื่อไปคำนวณหักลบใน Dashboard
    const category = loanType === "car_loan" ? "car_payment" : "home_payment";
    ss.getSheetByName("expenses").appendRow([
      Utilities.getUuid(), new Date(), category, payAmt,
      rows[i][3] || "",   
      `Loan Pay: ${rows[i][1]} ${interest > 0 ? `(ดอกเบี้ย ${interest}฿)` : ""}`,
      loanId, loanType, new Date()
    ]);

    // 3. บันทึกลง loan_payments เพื่อเก็บประวัติแยกต้น-ดอก
    if (paySheet) {
      paySheet.appendRow([
        Utilities.getUuid(), // id
        loanId,              // loan_id
        payAmt,              // amount (ยอดจ่ายรวม)
        principal,           // principal (เงินต้น)
        interest,            // interest (ดอกเบี้ย)
        new Date(),          // payment_date
        newPaidInst,         // installment_no
        new Date()           // created_at
      ]);
    }

    return output({ status: "success", remaining: newRemAmt, remaining_months: newRemMon });
  }
  return output({ error: "Loan not found" });
}

function getCreditSummary() {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const cardRows = ss.getSheetByName("credit_cards").getDataRange().getValues();
  const instRows = ss.getSheetByName("credit_installments").getDataRange().getValues();
  cardRows.shift(); instRows.shift();

  return cardRows.map(card => {
    const cardId = String(card[0]).trim();
    const limit  = Number(card[2]) || 0;
    const dueDay = Number(card[4]) || 0;

    let totalUnpaid = 0;
    let nextDueDate = null, nextDueAmt = 0, smallestNo = null;
    instRows.forEach(inst => {
      if (String(inst[2]).trim() !== cardId || inst[6] !== "unpaid") return;
      totalUnpaid += Number(inst[4]) || 0;
      const no = Number(inst[3]);
      if (smallestNo === null || no < smallestNo) {
        smallestNo = no; nextDueDate = inst[5]; nextDueAmt = Number(inst[4]) || 0;
      }
    });

    const utilization = limit > 0 ? round2((totalUnpaid / limit) * 100) : 0;
    return {
      card_id: cardId,
      name: card[1],
      credit_limit: limit,
      outstanding: round2(totalUnpaid),
      available_credit: round2(limit - totalUnpaid),
      utilization_percent: utilization,               // ← FIX: added
      next_due_date: nextDueDate ? fmt_date(nextDueDate) : null,  // ← FIX: added
      next_due_amount: nextDueAmt,                    // ← FIX: added
      closing_day: Number(card[3]) || 0,
      due_day: dueDay
    };
  });
}

function getCreditStatement(cardId, yearParam, monthParam) {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const cardRows = ss.getSheetByName("credit_cards").getDataRange().getValues();
  const instRows = ss.getSheetByName("credit_installments").getDataRange().getValues();
  cardRows.shift(); instRows.shift();

  const card = cardRows.find(r => String(r[0]).trim() === String(cardId).trim());
  if (!card) return { error: "Card not found" };

  const closingDay = Number(card[3]);
  const dueDay     = Number(card[4]);
  const year  = Number(yearParam);
  const month = Number(monthParam) - 1; // JS 0-indexed

  const periodEnd   = new Date(year, month, closingDay);
  const periodStart = new Date(year, month - 1, closingDay + 1);
  const dueDate     = new Date(year, month + 1, dueDay);

  let statementTotal = 0;
  instRows.forEach(inst => {
    if (String(inst[2]).trim() !== String(cardId).trim() || inst[6] !== "unpaid") return;
    const due = new Date(inst[5]);
    if (due > periodEnd && due <= dueDate) statementTotal += Number(inst[4]) || 0;
  });

  return {
    card_id: cardId,
    statement_period_start: fmt_date(periodStart),
    statement_period_end:   fmt_date(periodEnd),
    due_date: fmt_date(dueDate),
    statement_total: round2(statementTotal)
  };
}

function createCreditTransaction(data) {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  const txId = Utilities.getUuid();
  const now = new Date();
  
  // ✅ รับค่าจำนวนงวดที่จ่ายไปแล้ว
  const paidCount = Number(data.paid_installments) || 0;
  const inputMonths = Number(data.installment_months) || 1;
  const months = inputMonths <= 0 ? 1 : inputMonths;
  const amount = Number(data.amount) || 0;
  const perMonth = round2(amount / months);

  // 1. บันทึก Transaction หลัก
  ss.getSheetByName("credit_transactions").appendRow([
    txId, data.card_id, data.type || "purchase", data.description,
    amount, new Date(data.transaction_date), months, "active", now
  ]);

  // 2. คำนวณรอบบิล
  const cardRows = ss.getSheetByName("credit_cards").getDataRange().getValues();
  cardRows.shift();
  const card = cardRows.find(r => String(r[0]).trim() === String(data.card_id).trim());
  const closingDay = card ? Number(card[3]) : 25;
  const dueDay     = card ? Number(card[4]) : 15;

  const txDate = new Date(data.transaction_date);
  let statementMonth = txDate.getMonth() + (txDate.getDate() > closingDay ? 1 : 0);
  let dueMonth = statementMonth + (dueDay < closingDay ? 1 : 0);
  let dueYear  = txDate.getFullYear();
  
  if (dueMonth > 11) { dueMonth -= 12; dueYear++; }

  const instSheet = ss.getSheetByName("credit_installments");
  const expSheet  = ss.getSheetByName("expenses");

  // 3. สร้างงวดผ่อนชำระ (Loop งวดที่ระบุว่าจ่ายแล้วให้เป็น paid)
  for (let i = 0; i < months; i++) {
    let m = dueMonth + i, y = dueYear;
    while (m > 11) { m -= 12; y++; }
    const due = new Date(y, m, dueDay);
    
    // ✅ Logic มาร์คจ่ายแล้วย้อนหลัง
    const isPaid = i < paidCount;
    const status = isPaid ? "paid" : "unpaid";
    const payTimestamp = isPaid ? new Date() : "";

    instSheet.appendRow([Utilities.getUuid(), txId, data.card_id, i + 1, perMonth, due, status, payTimestamp]);

    // ✅ ถ้าจ่ายย้อนหลัง ให้ลงบัญชีรายจ่ายด้วยเพื่อให้ตัวเลขในชีตครบ
    if (isPaid) {
      expSheet.appendRow([
        Utilities.getUuid(), due, "credit_payment", perMonth, "Transfer", 
        `งวดที่ ${i+1}/${months} (ย้อนหลัง): ${data.description}`, 
        txId, "credit", new Date()
      ]);
    }
  }

  return output({ status: "created", transaction_id: txId });
}

function cancelCreditTransaction(data) {
  const ss  = SpreadsheetApp.openById(SHEET_ID);
  const txId = String(data.transaction_id).trim();

  // ลบ installments
  const instSheet = ss.getSheetByName("credit_installments");
  const instRows  = instSheet.getDataRange().getValues();
  for (let i = instRows.length - 1; i >= 1; i--) {
    if (String(instRows[i][1]).trim() === txId) instSheet.deleteRow(i + 1);
  }

  // ลบ expense ที่ผูกกับ transaction
  const expSheet = ss.getSheetByName("expenses");
  const expRows  = expSheet.getDataRange().getValues();
  for (let i = expRows.length - 1; i >= 1; i--) {
    if (String(expRows[i][6]).trim() === txId) expSheet.deleteRow(i + 1);
  }

  // ลบ transaction
  const txSheet = ss.getSheetByName("credit_transactions");
  const txRows  = txSheet.getDataRange().getValues();
  for (let i = 1; i < txRows.length; i++) {
    if (String(txRows[i][0]).trim() === txId) { txSheet.deleteRow(i + 1); break; }
  }
  return output({ status: "cancelled" });
}

function getDuePayments() {
  // const debts = getDebts(); // ❌ ปิดการดึง Debt
  const insts = getInstallments();
  const result = [];

  /*
  debts.forEach(d => {
    if (d.status !== "closed" && d.remaining_amount > 0) {

      }
    });

  ["car_loan", "home_loan"].forEach(type => {

      if (loan.status !== "closed" && loan.remaining_amount > 0) {
        result.push({ type, source_id: loan.id, title: loan.name, subtitle: `${loan.lender} · งวด ${loan.current_installment}/${loan.total_installments}`, amount: loan.monthly_due, due_date: "" });
      }
    });
  });

  return result;

function processPayment(data) {
  const { type, source_id, amount } = data;
  if (!amount || amount <= 0) return output({ error: "Invalid amount" });
  if (type === "debt")        return payDebt({ debt_id: source_id, amount });
  if (type === "installment") return payCreditInstallment({ installment_id: source_id });
  if (type === "car_loan" || type === "home_loan") return payLoan({ loan_id: source_id, loan_type: type, amount });
  return output({ error: "Unknown payment type: " + type });
}
